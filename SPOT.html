function drawCircle(x, y) { const outerRadius = 25; const innerRadius = 10; // Save current canvas state ctx.save(); // Draw outer circle ctx.beginPath(); ctx.arc(x, y, outerRadius, 0, Math.PI * 2); ctx.closePath(); // Cut out the inner circle ctx.arc(x, y, innerRadius, 0, Math.PI * 2, true); // Set styles ctx.fillStyle = 'red'; ctx.fill('evenodd'); // Restore canvas state ctx.restore(); }
